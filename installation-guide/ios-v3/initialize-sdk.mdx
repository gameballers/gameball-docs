# Initialize the SDK

Initialize Gameball at app startup, typically in your `AppDelegate` or `SceneDelegate`:

<CodeGroup>
```swift Swift (AppDelegate)
import UIKit
import Gameball

@main
class AppDelegate: UIResponder, UIApplicationDelegate {
    
    func application(
        _ application: UIApplication,
        didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?
    ) -> Bool {
        
        let config = GameballConfig(
            apiKey: "your-api-key",          // Required
            lang: "en",                      // Required
            platform: "ios",                 // Optional
            shop: "your-shop-id",            // Optional
            sessionToken: "your-session-token", // Optional - for v4.1 endpoints
            apiPrefix: "custom-prefix"       // Optional
        )
        
        Gameball.shared.initialize(with: config)
        
        return true
    }
}
```

```swift Swift (SwiftUI)
import SwiftUI
import Gameball

@main
struct YourApp: App {
    
    init() {
        let config = GameballConfig(
            apiKey: "your-api-key",
            lang: "en",
            platform: "ios",
            shop: "your-shop-id",
            sessionToken: "your-session-token"
        )
        
        Gameball.shared.initialize(with: config)
    }
    
    var body: some Scene {
        WindowGroup {
            ContentView()
        }
    }
}
```
</CodeGroup>

## Configuration Parameters

<ParamField body="apiKey" type="String" required>
Your Gameball API key from the dashboard (required).
</ParamField>

<ParamField body="lang" type="String" required>
Language code: "en", "ar", or other supported languages (required).
</ParamField>

<ParamField body="platform" type="String">
Platform identifier, defaults to "ios".
</ParamField>

<ParamField body="shop" type="String">
Your Gameball shop identifier for multi-shop configurations.
</ParamField>

<ParamField body="sessionToken" type="String">
Session Token for secure authentication (enables automatic v4.1 endpoint routing).
</ParamField>

<ParamField body="apiPrefix" type="String">
Custom API endpoint prefix for advanced configurations.
</ParamField>

### Validation Rules

**GameballConfig requires:**
- API key cannot be null or empty
- Language cannot be null or empty

## Configuration Examples

### Basic Setup

```swift
let config = GameballConfig(
    apiKey: "gb_live_xxxxxxxxxxxxxxxx",
    lang: "en"
)

Gameball.shared.initialize(with: config)
```

### Multi-Shop Configuration

```swift
let config = GameballConfig(
    apiKey: "your-api-key",
    lang: "en",
    shop: "shop_us"  // Different shop per region
)

Gameball.shared.initialize(with: config)
```

### With Session Token (v4.1)

```swift
let config = GameballConfig(
    apiKey: "your-api-key",
    lang: "en",
    sessionToken: "your-session-token"  // Enables v4.1 endpoints
)

Gameball.shared.initialize(with: config)
```

### Custom API Endpoint

```swift
let config = GameballConfig(
    apiKey: "your-api-key",
    lang: "en",
    apiPrefix: "https://custom-api.yourcompany.com"
)

Gameball.shared.initialize(with: config)
```

## Environment-Specific Configuration

Use different configurations for development and production:

```swift
import UIKit
import Gameball

@main
class AppDelegate: UIResponder, UIApplicationDelegate {
    
    func application(
        _ application: UIApplication,
        didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?
    ) -> Bool {
        
        #if DEBUG
        let config = GameballConfig(
            apiKey: "gb_test_xxxxxxxxxxxxxxxx",  // Test key
            lang: "en"
        )
        #else
        let config = GameballConfig(
            apiKey: "gb_live_xxxxxxxxxxxxxxxx",  // Production key
            lang: "en"
        )
        #endif
        
        Gameball.shared.initialize(with: config)
        
        return true
    }
}
```

## Using Configuration Files

Store configuration in a plist file:

**GameballConfig.plist:**
```xml
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
    <key>APIKey</key>
    <string>your-api-key</string>
    <key>Language</key>
    <string>en</string>
    <key>Shop</key>
    <string>your-shop-id</string>
</dict>
</plist>
```

**Load from plist:**
```swift
func loadGameballConfig() -> GameballConfig? {
    guard let path = Bundle.main.path(forResource: "GameballConfig", ofType: "plist"),
          let dict = NSDictionary(contentsOfFile: path) as? [String: Any],
          let apiKey = dict["APIKey"] as? String,
          let lang = dict["Language"] as? String else {
        return nil
    }
    
    return GameballConfig(
        apiKey: apiKey,
        lang: lang,
        shop: dict["Shop"] as? String,
        platform: "ios"
    )
}

// In AppDelegate
if let config = loadGameballConfig() {
    Gameball.shared.initialize(with: config)
}
```

<Tip>
Initialize the SDK as early as possible in your app's lifecycle to ensure it's ready when you need to track events or show the profile widget.
</Tip>

<Warning>
Make sure to use your actual API key from the Gameball dashboard. Test keys are for development, production keys are for live deployment. Never commit production keys to version control.
</Warning>
